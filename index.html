var existingArray = [{ "content": { "host": { "name": "Mia" } }, }, { "content": { "host": { "name": "Annie" } }, }, { "content": { "host": { "name": "Mia" } }, , { content: { host: { name: "Oscar" } }, }, { "content": { "host": { "name": "Annie" } }, }, { "content": { "host": { "name": "Mia" } }, }, { "content": { "host": { "name": "Annie" } }, } ];
How would I modify the object's name value if they match -- to differentiate them? To produce a new array, such as:
var existingArray = [{ "content": { "host": { "name": "Mia" } }, }, { "content": { "host": { "name": "Annie" } }, }, { "content": { "host": { "name": "Mia_2" } }, , { content: { host: { name: "Oscar" } }, }, { "content": { "host": { "name": "Annie_2" } }, }, { "content": { "host": { "name": "Mia_3" } }, }, { "content": { "host": { "name": "Annie_3" } }, } ];
